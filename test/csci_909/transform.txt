(type add [a a a] [(Num a)])
(define add (lambda [x y] (+ x y)))

(type z integer)
(define z (add 1 2))

(type pi double)
(define pi 3.14)

(record Pair [x y] [a b])
(type addPairs [(Pair a b) (Pair a b) (Pair a b)] [(Num a) (Num b)])
(define addPairs (lambda [p1 p2] (Pair (+ (Pair-x p1) (Pair-x p2)) (+ (Pair-y p1) (Pair-y p2)))))

(type pint (Pair a b))
(define pint (addPairs (Pair 1 2) (Pair 3 4)))

(type pdouble (Pair a b))
(define pdouble (addPairs (Pair 1 2.0) (Pair 3 4.0)))
