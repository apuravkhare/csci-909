(data (List a) (Cons [i c] [a (List a)]) (Empty))

(type e (List a))
(define e (Empty))

(type l (List a))
(define l (Cons 1 (Cons 2 (Empty))))

(type length [(List a) integer])
(define length (lambda [l] (cond
                             (Cons? l) (+ 1 (length (Cons-c l)))
                             (Empty? l) 0)))

(type sum [(List a) a] [(Num a)])
(define sum (lambda [l] (cond
                          (Cons? l) (+ (Cons-i l) (sum (Cons-c l)))
                          (Empty? l) 0)))

(length (Cons 1 (Cons 2 (Empty))))
(sum (Cons 1 (Cons 2 (Empty))))
(sum (Cons 1.0 (Cons 2.0 (Empty))))
